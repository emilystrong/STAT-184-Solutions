Exercise 8.2

library(ggplot2)
library(mosaicData)
data(CPS85)

Part (a)

CPS85 %>%
  ggplot(aes(x=wage)) +
  facet_wrap(.~married) +
  geom_density(aes(col=sex, fill=sex), alpha=0.5) +
  xlim(c(0,30)) +
  theme_bw()

Part (b)

CPS85 %>%
  ggplot(aes(x=age, y=wage)) +
  facet_wrap(~married, nrow=2) +
  geom_smooth(aes(shape=sex, col=sex)) +
  theme_bw()

Part (c)

CPS85 %>%
  ggplot(aes(x=age, y=wage)) +
  facet_wrap(~sector) +
  geom_smooth(aes(shape=sex, col=sex)) +
  ylim(c(0,25)) +
  theme_bw()
